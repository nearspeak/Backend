  sentence = I18n.t('errors.messages.not_saved',
                    count: resource.errors.count,
                    resource: resource.class.model_name.human.downcase)

    <a class="close" data-dismiss="alert" href="#">Ã—</a>
    <i class="icon-info-sign"></i>

    #{messages}
  </div>

  HTML
